import java.util.*;
class Instrument {
    void play() {
    }
}
class Piano extends Instrument {
    void play() {
        System.out.println("Piano is playing: Ting Ting");
    }
}
class Guitar extends Instrument {
    void play() {
        System.out.println("Guitar is playing: Strum Strum");
    }
}
class Drum extends Instrument {
    void play() {
        System.out.println("Drum is playing: Boom Boom");
    }
}
public class Main {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        System.out.println("Welcome to the Instrument Sound Simulator!");
        System.out.println("Select an instrument:");
        System.out.println("1. Piano");
        System.out.println("2. Guitar");
        System.out.println("3. Drum");
        System.out.print("Enter your choice (1-3): ");
        int choice = sc.nextInt();
        Instrument instrument;
        if (choice == 1) {
            instrument = new Piano();
            instrument.play();
        } 
        else if (choice == 2) {
            instrument = new Guitar();
            instrument.play();
        } 
        else if (choice == 3) {
            instrument = new Drum();
            instrument.play();
        } 
        else {
            System.out.println("Invalid choice.");
        }
        sc.close();
    }
}
